# FeaturesContactContact

Defines the supported contact features

## Example Usage

```typescript
import { FeaturesContactContact } from "crm/models/operations";

let value: FeaturesContactContact = {
  walletCodeFormat: "BARCODE",
  profile: [
    {
      isSupported: true,
    },
  ],
  communities: {
    isSupported: true,
  },
  countryCallingCodes: [
    {
      country: "CYP",
    },
  ],
  preferredLanguageCode: "EN",
};
```
